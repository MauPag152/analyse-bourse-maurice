packages = ["yfinance", "pandas", "matplotlib", "numpy"] import yfinance as yf import pandas as pd import numpy as np import matplotlib.pyplot as plt from js import document def update_chart(): ticker_symbol = document.getElementById("ticker-input").value.strip().upper() document.getElementById("status").innerHTML = "⌛ Calcul..." try: # 1. Récupération des données 1 an data = yf.download(ticker_symbol, period="1y") if data.empty: raise Exception("Ticker introuvable") # Correction format yfinance df = data['Close'].to_frame() if isinstance(data.columns, pd.MultiIndex) else data['Close'].to_frame() df = df.dropna() prices = df.values.flatten() last_price = float(prices[-1]) first_price = float(prices[0]) # 2. Calculs demandés yield_pct = ((last_price / first_price) - 1) * 100 daily_returns = df.pct_change().dropna() std_dev = float(daily_returns.std() * 100) # Écart-type journalier en % # 3. Mise à jour des tuiles de stats color_class = "text-green-400" if yield_pct >= 0 else "text-red-400" stats_html = f"""
Prix de Clôture
${last_price:.2f}
Rendement (1 an)
{yield_pct:+.2f}%
Écart-Type Journalier
{std_dev:.2f}%
""" document.getElementById("stats-container").innerHTML = stats_html # 4. Graphique 25x8 plt.style.use('dark_background') fig, ax = plt.subplots(figsize=(25, 8), facecolor='#1e293b') ax.set_facecolor('#1e293b') ax.plot(df.index, prices, color='#4ade80', linewidth=3, label=f'Cours {ticker_symbol}') ax.fill_between(df.index, prices, min(prices), color='#4ade80', alpha=0.1) ax.set_title(f"ÉVOLUTION ANNUELLE - {ticker_symbol}", fontsize=20, fontweight='bold', color='#f8fafc', pad=20) ax.grid(color='#334155', linestyle='--', alpha=0.5) ax.tick_params(colors='#94a3b8') document.getElementById("plot-output").innerHTML = "" display(fig, target="plot-output") document.getElementById("status").innerHTML = "● Prêt" except Exception as e: document.getElementById("status").innerHTML = "❌ Erreur" document.getElementById("plot-output").innerHTML = f"Erreur : {str(e)}" # Lancement automatique au démarrage update_chart()
